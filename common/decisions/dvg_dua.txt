dvg_picardy_reclaimed = {
	is_shown = {
		exists = c:DUA
		root = c:DUA
		NOT = { has_variable = dua_reclaim_picardy }
	}
    possible = {

		any_country = {
			AND = {
			country_has_primary_culture = cu:anglois
			owns_entire_state_region = STATE_PICARDY
			NOT = { has_variable = dua_reclaim_picardy }
			}
			
		}
	
	}

	when_taken = {
		add_loyalists = {
			value = 0.033
			culture = cu:french
		}
		add_loyalists = {
			value = 0.033
			culture = cu:occitan
		}
		add_loyalists = {
			value = 0.033
			culture = cu:british
		}
		#add_loyalists = {
		#	value = 0.05
		#	culture = cu:welsh
		#}
		#add_loyalists = {
		#	value = 0.05
		#	culture = cu:irish
		#}

		set_variable = dua_reclaim_picardy

    }

	ai_chance = {
		base = 100
	}

}
