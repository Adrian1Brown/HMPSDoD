force_single_market = {
	if = {
		limit = {
			is_subject = yes
			overlord = {
				market = prev.market
				NOT = {
					has_diplomatic_pact = {
						who = prev
						type = grant_own_market
					}
				}
			}
		}
		overlord = {
			create_diplomatic_pact = {
				country = prev
				type = grant_own_market
			}
		}
	}
}
